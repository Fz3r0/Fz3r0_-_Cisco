#! L3 LINK @ >> SERVER-2

exit

#! L3 LINK @ >> SERVER-3

   description ** Physical Link to -> SERVER-3 **
   no shutdown
   duplex full
      ip address  128.1.1.11 255.255.240.0
exit

#! L3 LINK @ >> SERVER-4

   description ** Physical Link to -> SERVER-4 **
   no shutdown
   duplex full
      ip address  111.0.0.2 255.255.255.252
exit

#! L3 LINK @ >> SERVER-5
interface Ethernet 1/1
   description ** Physical Link to -> SERVER-5 **
   no shutdown
   duplex full
      ip address  1.1.1.2 255.0.0.0
exit

!#######################################################################################
!#  L3 INTERNET INTERFACE @ ROUTER-2 (LAN/WAN)                                         #
!#######################################################################################

!# INTERNET INTERFACE @ ROUTER-2 (LAN/WAN) 
interface Ethernet 0/0
   description ** Link-to-ROUTER LAN/WAN **
   no shutdown
   duplex full
      ip address 123.1.1.1 255.255.255.252
         ip nat outside
exit
!# Default Route to Internet (ISP)
ip route 0.0.0.0 0.0.0.0 123.1.1.2

!#  SAVE CONFIGURATION                                                                 #
!#######################################################################################

#    LOGIN        : admin / admin.cisco                                               #
#                                                                                     #
#######################################################################################


$

!# NAMINGS, USERS, LICENSES, DISCOVERY
hostname IOS-ROUTER-2

!#######################################################################################
!#  ENABLE IP ROUTING                                                                  #
!#######################################################################################

!# Enables the router to forward packets BETWEEN DIFFERENT NETWORKS (turns on Layer 3 routing between management & default VRFs)
ip routing

!#######################################################################################
!#  PHYSICAL LINK & SUB-INTERFACES @ LAN (link to Switch)                              #
!#######################################################################################

#! L3 PHYSICAL INTERFACE @ GATEWAY FOR LAN :: VLANS: 10,20,30,66
#! Gateway Sub-Interfaces @ LAN
interface Ethernet 0/1
   description ** Physical Link to -> IOS-SWITCH **
   no shutdown
   duplex full
exit

#! L3 SUB-INTERFACE @ GATEWAY FOR - VLAN-66 MGMT (nat inside)
interface Ethernet 0/1.66
   description ** V66 Sub-Interface for -> VLAN-66 MGMT **
   no shutdown
      encapsulation dot1Q 66
      ip address 10.66.0.1 255.255.255.0
         ip nat inside
exit
#! L3 SUB-INTERFACE @ GATEWAY FOR - VLAN-10 BLUE (nat inside)
interface Ethernet 0/1.10
   description ** V10 Sub-Interface for -> VLAN-10 BLUE **
   no shutdown
      encapsulation dot1Q 10
      ip address 10.10.0.1 255.255.0.0
         ip nat inside
exit
#! L3 SUB-INTERFACE @ GATEWAY FOR - VLAN-20 RED (nat inside)
interface Ethernet 0/1.20
   description ** V20 Sub-Interface for -> VLAN-20 RED **
   no shutdown
      encapsulation dot1Q 20
      ip address 10.20.0.1 255.255.0.0
         ip nat inside
exit
#! L3 SUB-INTERFACE @ GATEWAY FOR - VLAN-30 GREEN (nat inside)
interface Ethernet 0/1.30
   description ** V30 Sub-Interface for -> VLAN-30 GREEN **
   no shutdown
      encapsulation dot1Q 30
      ip address 10.30.0.1 255.255.0.0
         ip nat inside
exit

!#######################################################################################
!#  L3 INTERFACE @ WAN + NAT TRANSLATION / INTERNET // OUTSIDE                         #
!#######################################################################################

!# WAN INTERFACE [Default Route @ Internet] (nat outside)
interface Ethernet 0/0
   description ** Link-to-INTERNET **
   no shutdown
   duplex full
      ip address 123.1.1.2 255.255.255.252
         ip nat outside
exit
!# Default Route to Internet (ISP)
ip route 0.0.0.0 0.0.0.0 123.1.1.1

!#######################################################################################
!# << ACL :: NAT >> NAT TRANSLATION CONFIGURATION, INSIDE-OUTSIDE                      #
!#######################################################################################

!# ACL que define el trafico interno a traducir
ip access-list extended NAT_INSIDE
   permit ip 10.0.0.0 0.255.255.255 any

!# Overload all matching inside traffic to the WAN interface address
ip nat inside source list NAT_INSIDE interface Ethernet 0/0 overload

!#######################################################################################
!#  SAVE CONFIGURATION                                                                 #
!#######################################################################################

end
write memory

!
!


````

#    MGMT         : 10.66.0.10/24                                                     #
#    LOGIN        : admin / admin.cisco                                               #
#                                                                                     #
#######################################################################################


$

!# NAMINGS, USERS, LICENSES, DISCOVERY, NO-DNS-RESOLVE

no ip domain-lookup

!#######################################################################################
!#  VLANS                                                                              #
!#######################################################################################

vlan 66
   name V66-MGMT
exit
vlan 10
   name V10-BLUE
exit
vlan 20
   name V20-RED
exit
vlan 30
   name V30-GREEN
exit

!#######################################################################################
!#  MANAGEMENT SVI & INTERFACE                                                         #
!#######################################################################################

!# INTERFACE :: SVI @ MGMT

interface vlan 66
   description ** SVI-V66-MGMT **
   no shutdown
      ip address 10.66.0.10 255.255.255.0 
exit
ip routing
ip route 0.0.0.0 0.0.0.0 10.66.0.1

!#######################################################################################
!#  WAN INTERFACE - TRUNK (allowed 10,20,30,66)                                        #
!#######################################################################################
!# INTERFACE :: Eth 0/0 @ LAN (TRUNK)
interface ethernet 0/0
   description ** TRUNK @ LAN **
   no shutdown
   duplex full
      switchport
      switchport trunk encapsulation dot1q
      switchport mode trunk
      switchport trunk allowed vlan 10,20,30,66
exit

!#######################################################################################
!#  LAN INTERFACES - ACCESS (10,20,30)                                                 #
!#######################################################################################

!# INTERFACES :: Eth 0/1 & 0/2 @ VLAN 10 BLUE
interface range ethernet0/1,ethernet0/2
   description ** ACCESS @ VLAN-10 BLUE **
   no shutdown
   duplex full
      switchport
      switchport mode access
      switchport access vlan 10
exit

!# INTERFACES :: Eth 0/3 & 1/0 @ VLAN 20 RED
interface range ethernet0/3,ethernet1/0
   description ** ACCESS @ VLAN-20 RED **
   no shutdown
   duplex full
      switchport
      switchport mode access
      switchport access vlan 20
exit

!# INTERFACES :: Eth 1/1 & 1/2 @ VLAN 30 GREEN
interface range ethernet1/1,ethernet1/2
   description ** ACCESS @ VLAN-30 GREEN **
   no shutdown
   duplex full
      switchport
      switchport mode access
      switchport access vlan 30
exit

!#######################################################################################
!#  SAVE CONFIGURATION                                                                 #
!#######################################################################################

end
write memory


!
!


````
